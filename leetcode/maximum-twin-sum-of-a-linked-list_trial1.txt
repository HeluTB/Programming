# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        nums = []
        cur = head
        while cur:
            nums.append(cur.val)
            cur = cur.next

        max_sum = 0
        n = len(nums)
        cur = head
        for i in range(n // 2):
            twin = nums.pop()
            max_sum = max(max_sum, cur.val + twin)
            cur = cur.next
        
        return max_sum
