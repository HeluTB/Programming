class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        target = n // 4
        counts = Counter(s)

        if all(counts[char] <= target for char in "QWER"): return 0

        i, res = 0, n
        for j in range(n):
            counts[s[j]] -= 1
            while i < n and all(counts[char] <= target for char in "QWER"):
                res = min(res, j - i + 1)
                counts[s[i]] += 1
                i += 1
        
        return res
