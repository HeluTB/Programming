class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        n = len(nums)
        cur_sum = sum(nums[:k])
        total = cur_sum

        for i in range(k, n):
            cur_sum += nums[i] - nums[i - k]
            if cur_sum > total:
                total = cur_sum
        
        return total / k
