class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l, h = 1, max(nums)
        res = h
        
        while l <= h:
            mid = (l + h) // 2
            cur_sum = 0
            for n in nums:
                cur_sum += (n + mid - 1) // mid
                
            if cur_sum <= threshold:
                res = mid
                h = mid - 1
            else:
                l = mid + 1
                
        return res
