# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def splitListToParts(self, head: Optional[ListNode], k: int) -> List[Optional[ListNode]]:
        cur = head
        size = 0
        while cur:
            size += 1
            cur = cur.next
        
        full = size // k
        extra = size % k

        nums = []
        cur = head
        for i in range(k):
            p = cur
            lim = 1 if i < extra else 0
            cur_part = full + lim

            for j in range(cur_part - 1):
                if cur: cur = cur.next
            
            if cur:
                nxt = cur.next
                cur.next = None
                cur = nxt
            
            nums.append(p)

        return nums
