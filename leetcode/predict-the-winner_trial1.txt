class Solution:
    def predictTheWinner(self, nums: list[int]) -> bool:
        memo = {}

        def solve(l, r):
            if l == r:
                return nums[l]
            
            if (l, r) in memo:
                return memo[(l, r)]
            
            left = nums[l] - solve(l + 1, r)
            right = nums[r] - solve(l, r - 1)
            
            memo[(l, r)] = max(left, right)
            return memo[(l, r)]

        return solve(0, len(nums) - 1) >= 0
