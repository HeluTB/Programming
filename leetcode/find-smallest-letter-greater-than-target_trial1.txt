class Solution:
    def nextGreatestLetter(self, letters: List[str], target: str) -> str:
        l, h = 0 , len(letters) - 1
        res = letters[0]

        if target >= letters[h]:
            return res

        while l <= h:
            mid = (l + h) // 2
            if letters[mid] > target:
                res = letters[mid]
                h = mid - 1
            else:
                l = mid + 1
        
        return letters[l]
