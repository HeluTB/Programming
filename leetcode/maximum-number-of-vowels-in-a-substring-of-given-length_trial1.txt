class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        n = len(s)
        vowels = {'a', 'e', 'i', 'o', 'u'}
        cur_vowels = 0
        
        for i in range(k):
            if s[i] in vowels:
                cur_vowels += 1
        
        max_vowels = cur_vowels
        for i in range(k, n):
            if s[i] in vowels:
                cur_vowels += 1
            
            if s[i - k] in vowels:
                cur_vowels -= 1

            max_vowels = max(max_vowels, cur_vowels)
            if max_vowels == k:
                return k
        
        return max_vowels
