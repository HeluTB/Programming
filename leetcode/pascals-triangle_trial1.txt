class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows == 1:
            return [[1]]
        
        triangle = self.generate(numRows - 1)
        prev_row = triangle[-1]
        n = len(prev_row)
        
        cur_row = [1]
        for i in range(n - 1):
            cur_row.append(prev_row[i] + prev_row[i + 1])
        cur_row.append(1)
        
        triangle.append(cur_row)
        return triangle
