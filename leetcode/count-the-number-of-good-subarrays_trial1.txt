class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        n = len(nums)
        count = Counter()
        cur_pairs = 0
        res = 0
        
        l = 0
        for r in range(n):
            cur_pairs += count[nums[r]]
            count[nums[r]] += 1

            while cur_pairs >= k:
                res += (n - r)
                count[nums[l]] -= 1
                cur_pairs -= count[nums[l]]
                l += 1
        
        return res
