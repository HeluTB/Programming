from collections import deque, Counter
class Solution:
    def firstUniqChar(self, s: str) -> int:
        count = Counter()
        queue = deque()
        
        for i, char in enumerate(s):
            count[char] += 1
            queue.append((char, i))
            
            while queue and count[queue[0][0]] > 1:
                queue.popleft()
        
        return queue[0][1] if queue else -1
