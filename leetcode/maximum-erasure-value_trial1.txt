class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        my_set = set()
        l, max_sum, cur_sum = 0, 0, 0
        for r in range(n):
            while nums[r] in my_set:
                my_set.remove(nums[l])
                cur_sum -= nums[l]
                l += 1
            
            my_set.add(nums[r])
            cur_sum += nums[r]
            max_sum = max(max_sum, cur_sum)
        
        return max_sum
