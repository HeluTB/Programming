class Solution:
    def isPalindrome(self, s: str) -> bool:
        s = s.lower()
        i, j = 0, len(s) - 1

        while i < j:
            while i < j and not ('a' <= s[i] <= 'z' or '0' <= s[i] <= '9'):
                i += 1
            while i < j and not ('a' <= s[j] <= 'z' or '0' <= s[j] <= '9'):
                j -= 1
            
            if s[i] != s[j]:
                return False
            i += 1
            j -= 1
        
        return True
