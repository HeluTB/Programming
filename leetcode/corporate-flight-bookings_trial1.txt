class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        diff = [0] * (n + 2)
        for l,r,s in bookings:
            diff[l] += s
            diff[r + 1] -= s

        cur_sum = 0
        for i in range(1, n + 1):
            diff[i] += diff[i - 1]
        
        return diff[1: n + 1]
