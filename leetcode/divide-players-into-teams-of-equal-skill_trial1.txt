class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort()
        n = len(skill)
        i, j = 0, n - 1

        target = skill[i] + skill[j]
        chemistry = 0

        while i < j:
            cur_sum = skill[i] + skill[j]
            if cur_sum != target:
                return -1
            
            chemistry += (skill[i] * skill[j])
            i += 1
            j -= 1
        
        return chemistry
