class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        l, high = 1, max(piles)
        res = high
        
        while l <= high:
            mid = (l + high) // 2
            total_hours = 0
            for p in piles:
                total_hours += (p + mid - 1) // mid
            
            if total_hours <= h:
                res = mid
                high = mid - 1
            else:
                l = mid + 1
                
        return res
