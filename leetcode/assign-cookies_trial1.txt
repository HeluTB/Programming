class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        n = len(g)
        m = len(s)
        i, j = 0, 0
        count = 0
        
        g.sort()
        s.sort()
        while i < n and j < m:
            if s[j] < g[i]:
                j += 1
            else:
                count += 1
                i += 1
                j += 1
        return count
