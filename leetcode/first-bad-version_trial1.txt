class Solution:
    def firstBadVersion(self, n: int) -> int:
        l, h = 1, n
        while l <= h:
            mid = l + (h - l) // 2
            if not isBadVersion(mid):
                l = mid + 1
            else:
                h = mid - 1
        
        return l
