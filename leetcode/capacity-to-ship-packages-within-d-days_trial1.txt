class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        l, h = max(weights), sum(weights)
    
        while l < h:
            mid = (l + h) // 2
            needed = 1
            cur_load = 0
            for w in weights:
                if cur_load + w > mid:
                    needed += 1
                    cur_load = w
                else:
                    cur_load += w
            
            if needed <= days:
                h = mid
            else:
                l = mid + 1
                
        return l
