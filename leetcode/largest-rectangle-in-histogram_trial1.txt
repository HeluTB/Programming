class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        heights.append(0)
        stack = [-1]
        area = 0
        n = len(heights)

        for i in range(n):
            while heights[i] < heights[stack[-1]]:
                h = heights[stack.pop()]
                w = i - stack[-1] - 1

                area = max(area, w * h)
            
            stack.append(i)
        
        heights.pop()
        return area
