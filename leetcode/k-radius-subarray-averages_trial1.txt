class Solution:
    def getAverages(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        window = (2 * k) + 1
        avgs = [-1] * n   
        if n < window:
            return avgs
        
        total = sum(nums[: window])
        for i in range(k, n - k):
            avgs[i] = total // window
            
            if i + k + 1 < n:
                total += nums[i + k + 1] - nums[i - k]

        return avgs
