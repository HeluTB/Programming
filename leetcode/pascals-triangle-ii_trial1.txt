class Solution:
    def getRow(self, rowIndex: int) -> list[int]:
        if rowIndex == 0:
            return [1]
        
        prev_row = self.getRow(rowIndex - 1)
        res = [1] * (rowIndex + 1)
        
        for i in range(1, rowIndex):
            res[i] = prev_row[i - 1] + prev_row[i]
            
        return res
