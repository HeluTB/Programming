class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        n = len(s)
        my_dict = {}
        i = 0
        count = 0

        for j in range(n):
            cur_char = s[j]
            if cur_char in my_dict and my_dict[cur_char] >= i:
                i = my_dict[cur_char] + 1
            
            my_dict[cur_char] = j
            count = max(count, j - i + 1)
                
        
        return count
